import pandas as pd
from transformers import CLIPProcessor, CLIPModel
from sentence_transformers import SentenceTransformer, util
from PIL import Image
import torch
import pytesseract
import cv2
from skimage.metrics import structural_similarity as ssim
from itertools import combinations

# Load Models
clip_model = CLIPModel.from_pretrained("openai/clip-vit-base-patch32")
clip_processor = CLIPProcessor.from_pretrained("openai/clip-vit-base-patch32")
text_model = SentenceTransformer('all-MiniLM-L6-v2')

# --- Functions ---
def extract_text(image_path):
    image = Image.open(image_path)
    return pytesseract.image_to_string(image)

def get_image_embedding(image_path):
    image = Image.open(image_path).convert("RGB")
    inputs = clip_processor(images=image, return_tensors="pt")
    with torch.no_grad():
        features = clip_model.get_image_features(**inputs)
    return features / features.norm(p=2, dim=-1, keepdim=True)

def compute_ssim(image_path1, image_path2):
    img1 = cv2.imread(image_path1, cv2.IMREAD_GRAYSCALE)
    img2 = cv2.imread(image_path2, cv2.IMREAD_GRAYSCALE)
    img1 = cv2.resize(img1, (min(img1.shape[1], img2.shape[1]), min(img1.shape[0], img2.shape[0])))
    img2 = cv2.resize(img2, (img1.shape[1], img1.shape[0]))
    score, _ = ssim(img1, img2, full=True)
    return score

def compute_text_similarity(text1, text2):
    emb1 = text_model.encode(text1, convert_to_tensor=True)
    emb2 = text_model.encode(text2, convert_to_tensor=True)
    return util.pytorch_cos_sim(emb1, emb2).item()

# CLIP

#LIP (Contrastive Language–Image Pretraining) is a model developed by OpenAI that understands images and text together. It can compute how well an image matches a piece of text — or #how similar two images are, even if they’re visually different but semantically the same.

#emantic and visual deduplication
#Understands layout + semantics: Even if a pie chart is in a different color or font, CLIP may still recognize it.


#How CLIP Works Step-by-Step: Image Encoder (ResNet or Vision Transformer) turns an image into a high-dimensional vector.

#Text Encoder (Transformer) turns text into a vector. During training, CLIP learns to align images with their correct captions and push unrelated pairs apart using a contrastive #loss.

#Once trained: Two similar images → vectors close together. Two dissimilar images → vectors far apart.

#0.9 to 1 - highly similar, 0.8 to 0.9 - Strong similarity, 0.70 to 0.79 - Moderate similarity, 0.60 to 0.69 - Weak similarity, <0.6 - Different content/layout/data

#SSIM ((Structural Similarity Index Measure))

#SSIM (Structural Similarity Index Measure) is a method to measure how visually similar two images are, focusing on structure, contrast, and luminanc#e, which better reflects human #perception than raw pixel comparison.

#How SSIM Works (in simple terms): Divides both images into small patches (e.g., 11×11).

#For each patch, it compares:

#Luminance (brightness),Contrast (variance), Structure (correlation)

#Combines these into a similarity score between 0 (different) and 1 (identical) using a formula: Averages all patch scores to get the final SSIM.

#✅ In Practice: SSIM ≈ 0.9–1.0 → Very similar images SSIM ≈ 0.5–0.8 → Some differences (layout, compression, charts) SSIM < 0.5 → Visually different##
